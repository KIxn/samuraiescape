377     /**
378      * Updates the various attributes of the character such as velocity,rotation etc
379      * @method
380      * @param {float} timeInSeconds -the number of seconds that the game has been running for
381      */
382     Update(timeInSeconds) {
383         if (!this._target) {
384             return;
385         }
386 
387         this._stateMachine.Update(timeInSeconds, this._input);
388 
389         const velocity = this._velocity;
390         const frameDecceleration = new THREE.Vector3(
391             velocity.x * this._decceleration.x,
392             velocity.y * this._decceleration.y,
393             velocity.z * this._decceleration.z
394         );
395         frameDecceleration.multiplyScalar(timeInSeconds);
396         frameDecceleration.z = Math.sign(frameDecceleration.z) * Math.min(
397             Math.abs(frameDecceleration.z), Math.abs(velocity.z));
398 
399         velocity.add(frameDecceleration);
400 
401         const controlObject = this._target;
402         const _Q = new THREE.Quaternion();
403         const _A = new THREE.Vector3();
404         const _R = controlObject.quaternion.clone();
405 
406         const acc = this._acceleration.clone();
407         if (this._input._keys.shift) {
408             //TODO adjust sprint speed
409             acc.multiplyScalar(6.0);
410         }
411 
412         if (this._stateMachine._currentState.Name == 'dance') {
413             acc.multiplyScalar(0.0);
414         }
415 
416         if (this._input._keys.forward) {
417             velocity.z += acc.z * timeInSeconds;
418         }
419         if (this._input._keys.backward) {
420             velocity.z -= acc.z * timeInSeconds;
421         }
422         if (this._input._keys.left) {
423             _A.set(0, 1, 0);
424             _Q.setFromAxisAngle(_A, 4.0 * Math.PI * timeInSeconds * this._acceleration.y);
425             _R.multiply(_Q);
426         }
427         if (this._input._keys.right) {
428             _A.set(0, 1, 0);
429             _Q.setFromAxisAngle(_A, 4.0 * -Math.PI * timeInSeconds * this._acceleration.y);
430             _R.multiply(_Q);
431         }
432 
433         controlObject.quaternion.copy(_R);
434 
435         const oldPosition = new THREE.Vector3();
436         oldPosition.copy(controlObject.position);
437 
438         const forward = new THREE.Vector3(0, 0, 1);
439         forward.applyQuaternion(controlObject.quaternion);
440         forward.normalize();
441 
442         const sideways = new THREE.Vector3(1, 0, 0);
443         sideways.applyQuaternion(controlObject.quaternion);
444         sideways.normalize();
445 
446         sideways.multiplyScalar(velocity.x * timeInSeconds);
447         forward.multiplyScalar(velocity.z * timeInSeconds);
448 
449         controlObject.position.add(forward);
450         controlObject.position.add(sideways);
451 
452         oldPosition.copy(controlObject.position);
453 
454         this._position.copy(controlObject.position);
455 
456         if (this._mixer) {
457             this._mixer.update(timeInSeconds);
458         }
459     }

581 /**
582  * Interface class for state machine to keep track of the character's movements
583  * @class
584  * Initiates State Machine
585  * @constructor
586  */
587 class FiniteStateMachine {
588 
589     constructor() {
590         this._states = {};
591         this._currentState = new IdleState(this);
592     }
593     /**
594      * Adds state to state machine
595      * @method
596      * @param {string} name - The name of the state
597      * @param {class} type - state class extending state
598      */
599     _AddState(name, type) {
600         console.log(type);
601         this._states[name] = type;
602     }
603 
604     /**
605      *transition from one state to the next
606      * @method
607      * @param {string} name - The name of the state
608      */
609     SetState(name) {
610         const prevState = this._currentState;
611 
612         if (prevState) {
613             if (prevState.Name == name) {
614                 return;
615             }
616             prevState.Exit();
617         }
618         /**
619          *Update the current state
620          * @method
621          * @param {number} timeElapsed - The number of seconds the game has been running for
622          * @param { Object} input - instance of a state (idle,run etc)
623          */
624 
625         Update(timeElapsed, input) {
626             if (this._currentState) {
627                 this._currentState.Update(timeElapsed, input);
628             }
629         }
630     };
631 
632     /**
633      * Class representing our characters Finite state machine
634      * @class
635      * @constructor
636      * @param { Object} proxy - instance of a finite state machine
637      * @extends FiniteStateMachine
638      */
639     class CharacterFSM extends FiniteStateMachine {
640 
641     constructor(proxy) {
642         super();
643         this._proxy = proxy;
644         this._Init();
645     }
646 
647     /**
648      * add states to our characters FSM
649      * @method
650      */
651     _Init() {
652         this._AddState('idle', IdleState);
653         this._AddState('walk', WalkState);
654         this._AddState('run', RunState);
655         this._AddState('dance', AttackState);
656     }
657 };

683 class AttackState extends State {
684     constructor(parent) {
685         super(parent);
686 
687         this._FinishedCallback = () => {
688             this._Finished();
689         }
690     }
691 
692     get Name() {
693         return 'dance';
694     }
695 
696     Enter(prevState) {
697         const curAction = this._parent._proxy._animations['dance'].action;
698         const mixer = curAction.getMixer();
699         mixer.addEventListener('finished', this._FinishedCallback);
700 
701         if (prevState) {
702             const prevAction = this._parent._proxy._animations[prevState.Name].action;
703 
704             curAction.reset();
705             curAction.setLoop(THREE.LoopOnce, 1);
706             curAction.clampWhenFinished = true;
707             curAction.crossFadeFrom(prevAction, 0.2, true);
708             curAction.play();
709         } else {
710             curAction.play();
711         }
712     }
713 
714     _Finished() {
715         this._Cleanup();
716         this._parent.SetState('idle');
717     }
718 
719     _Cleanup() {
720         const action = this._parent._proxy._animations['dance'].action;
721 
722         action.getMixer().removeEventListener('finished', this._CleanupCallback);
723     }
724 
725     Exit() {
726         this._Cleanup();
727     }
728 
729     Update(_) { }

733 class WalkState extends State {
734     constructor(parent) {
735         super(parent);
736     }
737 
738     get Name() {
739         return 'walk';
740     }
741 
742     Enter(prevState) {
743         const curAction = this._parent._proxy._animations['walk'].action;
744         if (prevState) {
745             const prevAction = this._parent._proxy._animations[prevState.Name].action;
746 
747             curAction.enabled = true;
748 
749             if (prevState.Name == 'run') {
750                 const ratio = curAction.getClip().duration / prevAction.getClip().duration;
751                 curAction.time = prevAction.time * ratio;
752             } else {
753                 curAction.time = 0.0;
754                 curAction.setEffectiveTimeScale(1.3);
755                 curAction.setEffectiveWeight(1.0);
756             }
757 
758             curAction.crossFadeFrom(prevAction, (prevAction.Name == 'walk') ? 0 : 0.5, true);
759             curAction.play();
760         } else {
761             curAction.play();
762         }
763     }
764 
765     Exit() { }
766 
767     Update(timeElapsed, input) {
768 
769         if (input._keys.forward || input._keys.backward) {
770             if (input._keys.shift) {
771                 this._parent.SetState('run');
772             }
773             return;
774         }
775         this._parent.SetState('idle');
776     }
777 };
